function ScoresPanel({user,updateUser}){
  const [score,setScore]=useState('');
  const [date,setDate]=useState(new Date().toISOString().split('T')[0]);
  const [editId,setEditId]=useState(null);
  const [editScore,setEditScore]=useState('');

  function addScore(){
    const s=parseInt(score);
    if(isNaN(s)||s<1||s>45){toast('Score must be between 1 and 45.','err');return;}
    const existing=user.scores||[];
    if(existing.find(x=>x.date===date)){toast('A score already exists for this date. Edit or delete the existing one.','err');return;}
    let newScores=[{id:`sc${Date.now()}`,score:s,date},... existing];
    newScores=newScores.sort((a,b)=>new Date(b.date)-new Date(a.date)).slice(0,5);
    updateUser({...user,scores:newScores});
    setScore('');toast('Score logged! ⛳','ok');
  }

  function deleteScore(id){
    const newScores=(user.scores||[]).filter(x=>x.id!==id);
    updateUser({...user,scores:newScores});
    toast('Score removed.','info');
  }

  function saveEdit(id){
    const s=parseInt(editScore);
    if(isNaN(s)||s<1||s>45){toast('Score must be between 1 and 45.','err');return;}
    const newScores=(user.scores||[]).map(x=>x.id===id?{...x,score:s}:x);
    updateUser({...user,scores:newScores});
    setEditId(null);toast('Score updated.','ok');
  }

  const scores=user.scores||[];
  return(
    <div className="fu">
      <div style={{display:'grid',gridTemplateColumns:'1fr 1fr',gap:20}}>
        <div className="card">
          <h3 style={{fontSize:15,fontWeight:600,marginBottom:4}}>Log a new score</h3>
          <p style={{fontSize:13,color:'var(--mu)',marginBottom:16}}>Enter a Stableford score (1–45). Only one score per date is allowed.</p>
          <div className="fg">
            <label>Stableford score</label>
            <input type="number" min="1" max="45" placeholder="e.g. 32" value={score} onChange={e=>setScore(e.target.value)}/>
          </div>
          <div className="fg">
            <label>Date played</label>
            <input type="date" value={date} onChange={e=>setDate(e.target.value)} max={new Date().toISOString().split('T')[0]}/>
          </div>
          <button className="btn btn-p" style={{width:'100%',justifyContent:'center'}} onClick={addScore}>Add score ⛳</button>
          {scores.length>=5&&<p style={{fontSize:12,color:'var(--amb)',marginTop:10}}>⚠ You have 5 scores. Adding a new one will replace the oldest.</p>}
        </div>
        <div className="card">
          <h3 style={{fontSize:15,fontWeight:600,marginBottom:4}}>Your score history</h3>
          <p style={{fontSize:13,color:'var(--mu)',marginBottom:16}}>Last 5 rounds · Most recent first</p>
          {scores.length===0?(
            <div style={{textAlign:'center',padding:'30px 0',color:'var(--mu)'}}>
              <div style={{fontSize:32,marginBottom:10}}>⛳</div>
              <div style={{fontSize:14}}>No scores yet. Log your first round!</div>
            </div>
          ):(
            <div style={{display:'flex',flexDirection:'column',gap:8}}>
              {scores.map((s,i)=>(
                <div key={s.id} style={{background:'var(--surf2)',border:'1px solid var(--bdr)',borderRadius:10,padding:'12px 14px',display:'flex',alignItems:'center',gap:12}}>
                  <div style={{fontFamily:'Syne,sans-serif',fontSize:26,fontWeight:700,color:'var(--amb)',width:44,textAlign:'center'}}>{s.score}</div>
                  <div style={{flex:1}}>
                    <div style={{fontSize:14,fontWeight:500}}>{fmtDate(s.date)}</div>
                    <div style={{fontSize:12,color:'var(--mu)'}}>Round {scores.length-i}</div>
                  </div>
                  {editId===s.id?(
                    <div style={{display:'flex',gap:6,alignItems:'center'}}>
                      <input type="number" min="1" max="45" value={editScore} onChange={e=>setEditScore(e.target.value)} style={{width:60,textAlign:'center',padding:'5px 8px'}}/>
                      <button className="btn btn-grn btn-sm" onClick={()=>saveEdit(s.id)}>✓</button>
                      <button className="btn btn-g btn-sm" onClick={()=>setEditId(null)}>✕</button>
                    </div>
                  ):(
                    <div style={{display:'flex',gap:6}}>
                      <button className="btn btn-o btn-sm" onClick={()=>{setEditId(s.id);setEditScore(s.score);}}>Edit</button>
                      <button className="btn btn-red btn-sm" onClick={()=>deleteScore(s.id)}>✕</button>
                    </div>
                  )}
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  );
}

function AdminDraws({users,draws,setDraws}){
  const [drawNums,setDrawNums]=useState(null);
  const [simulating,setSimulating]=useState(false);
  const [revealIdx,setRevealIdx]=useState(-1);
  const [mode,setMode]=useState('random');
  const activeSubs=users.filter(u=>u.sub?.status==='active').length;
  const pool=calcPool(activeSubs);
  const pendingDraw=draws.find(d=>d.status==='pending');

  function simulate(){
    setSimulating(true);
    setRevealIdx(-1);
    let nums;
    if(mode==='weighted'){
      // Weight by most common scores
      const allScores=users.filter(u=>u.sub?.status==='active').flatMap(u=>u.scores?.map(s=>s.score)||[]);
      const freq={};
      allScores.forEach(s=>{freq[s]=(freq[s]||0)+1;});
      // Pick top 5 most frequent
      const sorted=Object.entries(freq).sort((a,b)=>b[1]-a[1]);
      const pool=sorted.map(([k])=>+k);
      const picked=new Set();
      let i=0;
      while(picked.size<5&&i<pool.length){picked.add(pool[i]);i++;}
      while(picked.size<5)picked.add(ri(1,45));
      nums=Array.from(picked).sort((a,b)=>a-b);
    }else{
      nums=genDrawNums();
    }
    setDrawNums(nums);
    nums.forEach((_,i)=>{
      setTimeout(()=>setRevealIdx(i),600+(i*500));
    });
    setTimeout(()=>setSimulating(false),600+nums.length*500+200);
  }

  function publish(){
    if(!drawNums){toast('Run a simulation first.','err');return;}
    setDraws(prev=>prev.map(d=>d.status==='pending'?{...d,numbers:drawNums,status:'published',prizePool:pool}:d));
    toast('Draw published! Winners notified. 🎉','ok');
    setDrawNums(null);setRevealIdx(-1);
  }

  return(
    <div className="fu">
      <div style={{display:'grid',gridTemplateColumns:'1fr 1fr',gap:20}}>
        <div className="card">
          <h3 style={{fontSize:15,fontWeight:600,marginBottom:4}}>April 2026 draw</h3>
          <p style={{fontSize:13,color:'var(--mu)',marginBottom:16}}>Configure and simulate the monthly draw before publishing.</p>
          <div className="fg">
            <label>Draw mode</label>
            <select value={mode} onChange={e=>setMode(e.target.value)}>
              <option value="random">Random — standard lottery draw</option>
              <option value="weighted">Algorithmic — weighted by most frequent scores</option>
            </select>
          </div>
          <div style={{background:'var(--surf2)',borderRadius:10,padding:14,marginBottom:16}}>
            <div style={{display:'flex',justifyContent:'space-between',marginBottom:6}}><span style={{fontSize:13,color:'var(--mu)'}}>Active subscribers</span><span style={{fontWeight:600}}>{activeSubs}</span></div>
            <div style={{display:'flex',justifyContent:'space-between',marginBottom:6}}><span style={{fontSize:13,color:'var(--mu)'}}>Jackpot (5-match)</span><span style={{fontWeight:600,color:'var(--amb)'}}>£{fmt2(pool.jackpot)}</span></div>
            <div style={{display:'flex',justifyContent:'space-between',marginBottom:6}}><span style={{fontSize:13,color:'var(--mu)'}}>4-match prize</span><span style={{fontWeight:600}}>£{fmt2(pool.four)}</span></div>
            <div style={{display:'flex',justifyContent:'space-between'}}><span style={{fontSize:13,color:'var(--mu)'}}>3-match prize</span><span style={{fontWeight:600}}>£{fmt2(pool.three)}</span></div>
          </div>
          <div style={{display:'flex',gap:10}}>
            <button className="btn btn-o" style={{flex:1,justifyContent:'center'}} onClick={simulate} disabled={simulating}>{simulating?'Drawing...':'Run simulation'}</button>
            <button className="btn btn-p" style={{flex:1,justifyContent:'center'}} onClick={publish} disabled={!drawNums||simulating}>Publish draw ✦</button>
          </div>
        </div>
        <div className="card">
          <h3 style={{fontSize:15,fontWeight:600,marginBottom:16}}>Draw result {drawNums?'(simulation)':''}</h3>
          <div style={{display:'flex',gap:10,justifyContent:'center',marginBottom:20,minHeight:60,alignItems:'center'}}>
            {drawNums?drawNums.map((n,i)=>(
              <div key={n} className={`ball ${i<=revealIdx?'rev':''}`}>{i<=revealIdx?n:'?'}</div>
            )):(
              <div style={{color:'var(--mu2)',fontSize:14,textAlign:'center'}}>
                <div style={{fontSize:32,marginBottom:8}}>🎯</div>
                Run a simulation to preview the draw numbers.
              </div>
            )}
          </div>
          {drawNums&&revealIdx>=4&&(
            <div className="fu">
              <hr className="sep"/>
              <h4 style={{fontSize:14,fontWeight:600,marginBottom:12}}>Preliminary results</h4>
              {users.filter(u=>u.sub?.status==='active'&&(u.scores||[]).length===5).map(u=>{
                const m=checkMatch(u.scores,drawNums);
                if(m<3)return null;
                return(
                  <div key={u.id} style={{display:'flex',justifyContent:'space-between',alignItems:'center',padding:'8px 0',borderBottom:'1px solid var(--bdr)'}}>
                    <span style={{fontSize:14,fontWeight:500}}>{u.name}</span>
                    <span className={`badge ${m>=5?'badge-amb':m===4?'badge-blu':'badge-grn'}`}>{m}-match {m>=5?'🏆':m===4?'🥈':'🥉'}</span>
                  </div>
                );
              })}
              {users.filter(u=>u.sub?.status==='active'&&(u.scores||[]).length===5&&checkMatch(u.scores,drawNums)>=3).length===0&&(
                <p style={{fontSize:14,color:'var(--mu)'}}>No winners this draw — jackpot rolls over.</p>
              )}
            </div>
          )}
        </div>
      </div>
      <div style={{marginTop:20}}>
        <h3 style={{fontSize:15,fontWeight:600,marginBottom:14}}>Draw history</h3>
        <div style={{display:'flex',flexDirection:'column',gap:10}}>
          {draws.map(d=>(
            <div key={d.id} style={{background:'var(--surf2)',border:'1px solid var(--bdr)',borderRadius:10,padding:'14px 16px',display:'flex',alignItems:'center',gap:16,flexWrap:'wrap'}}>
              <div style={{minWidth:130}}>
                <div style={{fontWeight:600,fontSize:14}}>{d.month}</div>
                <span className={`badge ${d.status==='published'?'badge-grn':'badge-amb'}`} style={{marginTop:4}}>{d.status}</span>
              </div>
              <div style={{display:'flex',gap:6}}>
                {d.numbers?d.numbers.map(n=><div key={n} className="ball rev" style={{width:36,height:36,fontSize:13}}>{n}</div>):<span style={{color:'var(--mu)',fontSize:13}}>Not drawn yet</span>}
              </div>
              {d.prizePool?.total>0&&<div style={{marginLeft:'auto',textAlign:'right'}}><div style={{fontSize:12,color:'var(--mu)'}}>Prize pool</div><div style={{fontWeight:700,fontFamily:'Syne,sans-serif',color:'var(--amb)'}}>£{fmt2(d.prizePool.total)}</div></div>}
            </div>
          ))}
        </div>
      </div>
    </div>
  );
}

          <button className="btn btn-grn" onClick={add}>Add charity</button>
        </div>
      )}
      <div style={{display:'grid',gridTemplateColumns:'repeat(auto-fill,minmax(280px,1fr))',gap:14}}>
        {charities.map(c=>(
          <div key={c.id} className="card">
            <div style={{display:'flex',justifyContent:'space-between',alignItems:'flex-start',marginBottom:10}}>
              <div style={{fontSize:28}}>{c.emoji}</div>
              <button className="btn btn-red btn-sm" onClick={()=>remove(c.id)}>Remove</button>
            </div>
            <div style={{fontSize:11,color:'var(--grn)',fontWeight:600,marginBottom:4}}>{c.category}</div>
            <div style={{fontWeight:600,marginBottom:4}}>{c.name}</div>
            <div style={{fontSize:13,color:'var(--mu)',marginBottom:8,lineHeight:1.5}}>{c.desc}</div>
            <span className="badge badge-grn">{c.impact}</span>
          </div>
        ))}
      </div>
    </div>
  );
}

function AdminWinners({users,setUsers,draws}){
  const winners=users.flatMap(u=>(u.drawResults||[]).filter(r=>r.prize>0).map(r=>({...r,user:u})));
  function updateStatus(userId,drawId,status){
    setUsers(prev=>prev.map(u=>{
      if(u.id!==userId)return u;
      return{...u,
        drawResults:(u.drawResults||[]).map(r=>r.drawId===drawId?{...r,verifyStatus:status}:r),
        winnings:status==='paid'?{...u.winnings,pending:0,paid:(u.winnings?.paid||0)+(u.drawResults||[]).find(r=>r.drawId===drawId)?.prize||0}:u.winnings
      };
    }));
    toast(status==='paid'?'Payout marked as paid ✓':'Status updated.','ok');
  }
  return(
    <div className="fu">
      <div className="card">
        <h3 style={{fontSize:15,fontWeight:600,marginBottom:16}}>Winners & verification</h3>
        {winners.length===0?(
          <p style={{color:'var(--mu)',fontSize:14}}>No winners to display.</p>
        ):(
          <table>
            <thead><tr><th>Winner</th><th>Draw</th><th>Match</th><th>Prize</th><th>Verified</th><th>Actions</th></tr></thead>
            <tbody>
              {winners.map((w,i)=>(
                <tr key={i}>
                  <td style={{fontWeight:500}}>{w.user.name}</td>
                  <td style={{color:'var(--mu)',fontSize:13}}>{w.drawId==='d1'?'Feb 2026':w.drawId==='d2'?'Mar 2026':'Apr 2026'}</td>
                  <td><span className={`badge ${w.matched>=5?'badge-amb':w.matched===4?'badge-blu':'badge-grn'}`}>{w.matched}-match</span></td>
                  <td style={{fontFamily:'Syne,sans-serif',fontWeight:700,color:'var(--amb)'}}>£{fmt2(w.prize)}</td>
                  <td><span className={`badge ${w.user.verified?'badge-grn':'badge-red'}`}>{w.user.verified?'Verified':'Unverified'}</span></td>
                  <td>
                    <div style={{display:'flex',gap:6}}>
                      {!w.user.verified&&<button className="btn btn-o btn-sm" onClick={()=>{setUsers(p=>p.map(u=>u.id===w.user.id?{...u,verified:true}:u));toast('User verified ✓','ok');}}>Verify</button>}
                      {w.verifyStatus!=='paid'&&<button className="btn btn-grn btn-sm" onClick={()=>updateStatus(w.user.id,w.drawId,'paid')}>Mark paid</button>}
                      {w.verifyStatus==='paid'&&<span className="badge badge-grn">✓ Paid</span>}
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        )}
      </div>
    </div>
  );
}

function AdminReports({users,draws}){
  const active=users.filter(u=>u.sub?.status==='active').length;
  const inactive=users.filter(u=>u.sub?.status==='inactive').length;
  const monthly=users.filter(u=>u.sub?.plan==='monthly'&&u.sub?.status==='active').length;
  const yearly=users.filter(u=>u.sub?.plan==='yearly'&&u.sub?.status==='active').length;
  const totalWon=users.reduce((a,u)=>a+(u.winnings?.total||0),0);
  const totalCharity=users.filter(u=>u.sub?.status==='active').reduce((a,u)=>{
    const fee=subFee(u.sub?.plan);const pct=(u.charity?.pct||10)/100;
    return a+fee*pct;
  },0);
  const pool=calcPool(active);
  const published=draws.filter(d=>d.status==='published').length;

  const charityBreakdown=CHARITIES.map(c=>{
    const count=users.filter(u=>u.charity?.id===c.id&&u.sub?.status==='active').length;
    return{...c,count,total:users.filter(u=>u.charity?.id===c.id&&u.sub?.status==='active').reduce((a,u)=>a+(subFee(u.sub?.plan)*((u.charity?.pct||10)/100)),0)};
  }).filter(c=>c.count>0).sort((a,b)=>b.count-a.count);

  return(
    <div className="fu">
      <div style={{display:'grid',gridTemplateColumns:'repeat(auto-fit,minmax(160px,1fr))',gap:14,marginBottom:24}}>
        <div className="sc"><div className="sc-lbl">Total users</div><div className="sc-val" style={{color:'var(--tx)'}}>{users.length}</div></div>

        <div className="sc"><div className="sc-lbl">Draws completed</div><div className="sc-val" style={{color:'var(--tx)'}}>{published}</div></div>
      </div>
      <div style={{display:'grid',gridTemplateColumns:'1fr 1fr',gap:20}}>
        <div className="card">
          <h3 style={{fontSize:15,fontWeight:600,marginBottom:16}}>Subscription breakdown</h3>
          <div style={{marginBottom:12}}>
            <div style={{display:'flex',justifyContent:'space-between',marginBottom:6}}><span style={{fontSize:14}}>Monthly plan</span><span style={{fontWeight:600}}>{monthly}</span></div>
            <div className="pbar"><div className="pfill" style={{width:`${active?monthly/active*100:0}%`,background:'var(--amb)'}}/></div>
          </div>
          <div style={{marginBottom:12}}>
            <div style={{display:'flex',justifyContent:'space-between',marginBottom:6}}><span style={{fontSize:14}}>Yearly plan</span><span style={{fontWeight:600}}>{yearly}</span></div>

/* ───── ROOT APP ───── */
function App(){
  const [users,setUsers]=useState(()=>{
    try{const s=localStorage.getItem('orbit_users');return s?JSON.parse(s):INITIAL_USERS;}
    catch{return INITIAL_USERS;}
  });
  const [draws,setDraws]=useState(DRAWS);
  const [page,setPage]=useState('landing');
  const [authMode,setAuthMode]=useState(null);
  const [user,setUser]=useState(null);

  useEffect(()=>{
    try{const s=localStorage.getItem('orbit_session');if(s){const uid=JSON.parse(s);const u=users.find(x=>x.id===uid);if(u)setUser(u);}}catch{}
  },[]);

  function handleAuth(mode){setAuthMode(mode);}

  function handleAuthSuccess(newUser,newUsers){
    if(newUsers)setUsers(newUsers);
    setUser(newUser);
    setAuthMode(null);
    try{localStorage.setItem('orbit_session',JSON.stringify(newUser.id));}catch{}
    setPage(newUser.role==='admin'?'admin':'dashboard');
    toast(`Welcome${newUser.role==='admin'?' back, Admin':`, ${newUser.name.split(' ')[0]}`}! 🎉`,'ok');
  }

  function handleLogout(){
    setUser(null);setPage('landing');
    try{localStorage.removeItem('orbit_session');}catch{}
    toast('Signed out.','info');
  }

  function updateUser(updated){
    setUser(updated);
    setUsers(prev=>{
      const next=prev.map(u=>u.id===updated.id?updated:u);
      try{localStorage.setItem('orbit_users',JSON.stringify(next));}catch{}
      return next;
    });
  }

  function setDrawsAndPersist(fn){
    setDraws(typeof fn==='function'?fn:()=>fn);
  }

  return(
    <>
      <ToastSystem/>
      <Navbar page={page} setPage={p=>{if(!user&&p!=='landing'){setAuthMode('login');return;}setPage(p);}} user={user} onAuth={handleAuth} onLogout={handleLogout}/>
      {page==='landing'&&<Landing onAuth={handleAuth}/>}
      {page==='dashboard'&&user&&<Dashboard user={user} updateUser={updateUser} draws={draws} users={users}/>}
      {page==='charities'&&<CharitiesPage user={user} updateUser={updateUser}/>}
      {page==='admin'&&user?.role==='admin'&&<AdminPanel users={users} setUsers={u=>{const next=typeof u==='function'?u(users):u;setUsers(next);try{localStorage.setItem('orbit_users',JSON.stringify(next));}catch{}}} draws={draws} setDraws={setDrawsAndPersist}/>}
      {authMode&&<AuthModal mode={authMode} onClose={()=>setAuthMode(null)} onSuccess={handleAuthSuccess}/>}
    </>
  );
}

ReactDOM.createRoot(document.getElementById('root')).render(<App/>);
</script>
